            <p class="text-xs text-gray-400 uppercase tracking-widest">每月血汗錢總計</p>
            <h2 id="total-cost" class="text-5xl font-bold my-2 text-white">$ 0</h2>
            <p id="roast-summary" class="text-sm roast-text">正在掃描你的財務漏洞...</p>

        <div class="card rounded-xl p-4 mb-8">
            <h3 class="text-lg font-bold mb-4 text-white">➕ 增加新的供奉項目</h3>
            <div class="space-y-3">
                <input id="sub-name" type="text" placeholder="服務名稱 (如: Netflix)" class="w-full bg-black border border-gray-700 p-2 rounded focus:border-green-500 outline-none text-white">
                <input id="sub-price" type="number" placeholder="每月金額 (TWD)" class="w-full bg-black border border-gray-700 p-2 rounded focus:border-green-500 outline-none text-white">
                <button onclick="addSubscription()" class="w-full bg-green-600 hover:bg-green-500 text-black font-bold py-2 rounded transition">承認我是盤子並儲存</button>
            </div>

    <script>
        let subs = JSON.parse(localStorage.getItem('my_subscriptions')) || [];
        const roasts = ["這筆錢夠買 3 份排骨飯了！", "偵測到盤子行為，感謝貢獻。", "上次打開它是什麼時候？", "這筆扣款讓你離財富自由又遠了。", "你家沙發是鑲金的嗎？", "這是在買心理安慰吧？", "30元App你嫌貴，這個你付得很爽？"];
        window.onload = render;
        function addSubscription() {
            const nameEl = document.getElementById('sub-name');
            const priceEl = document.getElementById('sub-price');
            if (!nameEl.value || !priceEl.value) { alert("欄位填好，別想逃避！"); return; }
            const roast = roasts[Math.floor(Math.random() * roasts.length)];
            subs.push({ id: Date.now(), name: nameEl.value, price: parseFloat(priceEl.value), roast });
            saveData(); nameEl.value = ''; priceEl.value = ''; render();
        }
        function removeSub(id) { if(confirm("確定要取消供奉？")) { subs = subs.filter(s => s.id !== id); saveData(); render(); } }
        function saveData() { localStorage.setItem('my_subscriptions', JSON.stringify(subs)); }
        function render() {
            const listEl = document.getElementById('sub-list');
            const totalEl = document.getElementById('total-cost');
            const summaryEl = document.getElementById('roast-summary');
            listEl.innerHTML = ''; let total = 0;
            subs.forEach(sub => {
                total += sub.price;
                listEl.innerHTML += `<div class="card rounded-xl p-4 relative"><button onclick="removeSub(${sub.id})" class="absolute top-2 right-2 text-gray-600">✕</button><div class="flex justify-between items-center mb-2"><span class="text-white font-bold text-lg">${sub.name}</span><span class="text-xl">$${sub.price}</span></div><p class="text-xs roast-text">💬 ${sub.roast}</p></div>`;
            });
            totalEl.innerText = `$ ${total.toLocaleString()}`;
            summaryEl.innerText = total === 0 ? "目前帳面乾淨。" : (total > 1500 ? "錢包正在噴血！" : "小傷口還能救。");
        }
    </script>
